convert.ts 
import { fromBinary, fitReader } from '@kaiord/fit'
import { toText, tcxWriter } from '@kaiord/tcx'

const krd  = await fromBinary(fitFile, fitReader)
const tcx  = await toText(krd, tcxWriter)
